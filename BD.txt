
USE railway;

CREATE TABLE administrador (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    nombre_usuario VARCHAR(50) UNIQUE NOT NULL,
    contrasena VARCHAR(255) NOT NULL,
    nombre_completo VARCHAR(150) NOT NULL,
    correo_electronico VARCHAR(150) NOT NULL,
    rol VARCHAR(50) DEFAULT 'ADMIN',
    activo ENUM('activo', 'inactivo') DEFAULT 'activo',
    ultimo_acceso DATETIME NULL
);

CREATE TABLE usuario (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    tipo_persona VARCHAR(50) NOT NULL, -- NATURAL | JURIDICA | ANONIMA
    nombre_completo VARCHAR(150) NULL,
    documento_identidad VARCHAR(50) NULL,
    tipo_documento VARCHAR(20) NULL, -- CC | CE | TI | NIT | PASAPORTE
    correo_electronico VARCHAR(150) NULL,
    telefono VARCHAR(20) NULL,
    razon_social VARCHAR(150) NULL,
    nit VARCHAR(50) NULL,
    nombre_representante VARCHAR(150) NULL,
    correo_corporativo VARCHAR(150) NULL,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE configuracion_sistema (
    id BIGINT PRIMARY KEY DEFAULT 1,
    dias_vencimiento_peticion INTEGER DEFAULT 15,
    dias_vencimiento_queja INTEGER DEFAULT 15,
    dias_vencimiento_reclamo INTEGER DEFAULT 15,
    dias_vencimiento_sugerencia INTEGER DEFAULT 15,
    dias_vencimiento_denuncia INTEGER DEFAULT 15,
    correo_notificaciones VARCHAR(150) NULL,
    nombre_empresa VARCHAR(150) NULL
);

CREATE TABLE pqrs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    codigo_radicado VARCHAR(50) UNIQUE NOT NULL,
    tipo_solicitud VARCHAR(50) NOT NULL, -- PETICION | QUEJA | RECLAMO | SUGERENCIA | DENUNCIA
    asunto VARCHAR(255) NOT NULL,
    descripcion TEXT NOT NULL,
    archivo_adjunto VARCHAR(255) NULL,
    estado VARCHAR(50) DEFAULT 'PENDIENTE', -- PENDIENTE | EN_PROCESO | RESUELTO | RECHAZADO
    fecha_radicacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    fecha_vencimiento DATE NULL,
    respuesta_administrador TEXT NULL,
    fecha_respuesta DATETIME NULL,
    usuario_id BIGINT NOT NULL,
    administrador_id BIGINT NULL,
    FOREIGN KEY (usuario_id) REFERENCES usuario(id) ON DELETE CASCADE,
    FOREIGN KEY (administrador_id) REFERENCES administrador(id) ON DELETE SET NULL
);

CREATE TABLE historial_accion (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    pqrs_id BIGINT NOT NULL,
    administrador_id BIGINT NULL,
    accion_realizada VARCHAR(100) NOT NULL,
    estado_anterior VARCHAR(50) NULL,
    estado_nuevo VARCHAR(50) NULL,
    descripcion TEXT NULL,
    fecha_hora DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (pqrs_id) REFERENCES pqrs(id) ON DELETE CASCADE,
    FOREIGN KEY (administrador_id) REFERENCES administrador(id) ON DELETE SET NULL
);

CREATE TABLE alerta_vencimiento (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    pqrs_id BIGINT NOT NULL,
    dias_restantes INTEGER NOT NULL,
    nivel_alerta VARCHAR(20) NOT NULL, 
    notificacion_enviada BOOLEAN DEFAULT FALSE,
    fecha_generacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (pqrs_id) REFERENCES pqrs(id) ON DELETE CASCADE
);

CREATE TABLE reporte (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    fecha_generacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    tipo_reporte VARCHAR(100) NOT NULL,
    fecha_inicio DATE NULL,
    fecha_fin DATE NULL,
    total_recibidas INTEGER DEFAULT 0,
    total_resueltas INTEGER DEFAULT 0,
    total_pendientes INTEGER DEFAULT 0,
    total_rechazadas INTEGER DEFAULT 0,
    tiempo_promedio_respuesta DOUBLE DEFAULT 0,
    porcentaje_cumplimiento DOUBLE DEFAULT 0,
    formato_exportacion VARCHAR(20) NOT NULL, 
    administrador_id BIGINT NULL,
    FOREIGN KEY (administrador_id) REFERENCES administrador(id) ON DELETE SET NULL
);
